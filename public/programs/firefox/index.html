const https = require('http');

https.get('http://floshodan.io/api/about', (resp) => {
  let data = '';

  resp.on('data', (chunk) => {
    data += chunk;
  });

  resp.on('end', () => {
    console.log(JSON.parse(data).explanation);
  });

  // Handle error
}).on("error", (err) => {
  console.log("Error: " + err.message);
});
import requests

response = requests.get('http://floshodan.io/api/about')
json_response = response.json()
print(json_response)
